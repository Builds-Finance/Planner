function calc() { let B = Number(document.getElementById("lot").value); // Lot Size let F = Number(document.getElementById("entry").value); // Entry if (!B || !F) return; // ===== EXACT ORDER (same as Excel dependency) ===== // D = -4000 / (F * B) let D = -4000 / (F * B); // E = F * D + F let E = (F * D) + F; // G = AVERAGE(E, F) let G = (E + F) / 2; // C = 3000 / (G * B) let C = 3000 / (G * B); // H = (G * C) + G let H = (G * C) + G; // I = (H - F) * B let I = (H - F) * B; // J = E - F let J = E - F; // K = H - G let K = H - G; // ===== DISPLAY (important precision) ===== document.getElementById("rsiProfit").innerText = C.toFixed(9); document.getElementById("firstAvgPercent").innerText = D.toFixed(9); document.getElementById("firstAvg").innerText = E.toFixed(2); document.getElementById("avgPrice").innerText = G.toFixed(2); document.getElementById("rsiTarget").innerText = H.toFixed(2); document.getElementById("takeProfit").innerText = I.toFixed(0); document.getElementById("avgDiff").innerText = J.toFixed(2); document.getElementById("targetDiff").innerText = K.toFixed(2); }
